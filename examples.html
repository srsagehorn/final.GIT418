Example 1

Here we have more objects created

class FamilyMembers {
                constructors(name, age, relation) {
                      this.name = name;
                      this.age = age;
                      this.relation = relation;
                        }
                      }

let auntBetty = new FamilyMembers("Betty", 57, "aunt")
let uncleRoger = new FamilyMembers("Roger", 58, "Uncle")
let cousinJen = new FamilyMembers("Jen", 23, "Cousin")
let stepmomJulie = new FamilyMembers("Julie", 46, "Step Mom")

console.log (auntBetty, uncleRoger, cousinJen, stepmomJulie)
{name : "Betty", age : 57, relation : "Aunt"}
{name : "Roger", age : 58, relation : "Uncle"}
{name : "Jen", age : 23, relation : "Cousin"}
{name : "Julie", age : 46, relation : "Step Mom"}

Example 2

Here we have more constructors

class Animals {
                constructors(name, type, covering, habitat, color, aquatic, eater) {
                      this.name = name;
                      this.type = type;
                      this.covering = covering;
                      this.habitat = habitat;
                      this.color = color;
                      this.aquatic = aquatic;
                      this.eater = eater;
                        }
                      }

let pelican = new Animals("Pelican", "Bird", "Feathers", "Coastal", "White and Grey", false, "Carnivore")
let horse = new Animals("Horse", "Mammal", "Fur", "Grasslands", "Varied", false, "Herbivore")
let turtle = new Animals("Turtle", "Reptile", "Shell", "Shallow Water", "Green", true, "Herbivore")

console.log (pelican, horse, turtle)
{name : "Pelican",
type : "Bird",
covering : "Feathers",
habitat : "Coastal",
color : "White and Grey",
aquatic : false,
eater : "Carnivore"}

{name : "Horse",
type : "Mammal",
covering : "Fur",
habitat : "Grasslands",
color : "Varied",
aquatic : false,
eater : "Herbivore"}

{name : "Turtle",
